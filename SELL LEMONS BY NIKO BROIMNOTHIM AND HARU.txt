local function gradient(t)
    local r = ""
    local c = {
        Color3.fromRGB(243, 234, 255),
        Color3.fromRGB(196, 181, 253),
        Color3.fromRGB(139,  92, 246),
        Color3.fromRGB(109,  40, 217),
        Color3.fromRGB( 91,  63, 166),
    }
    local clean = t:gsub("[^%w]", "")
    local len   = #clean
    local index = 1
    for i = 1, #t do
        local ch = t:sub(i,i)
        if ch:match("[%w]") then
            local x    = (index-1) / math.max(len-1, 1)
            local pos  = x * (#c - 1)
            local low  = math.clamp(math.floor(pos)+1, 1, #c)
            local high = math.clamp(low+1, 1, #c)
            local c1, c2 = c[low], c[high]
            local blend  = pos % 1
            local rR = c1.R + (c2.R-c1.R)*blend
            local rG = c1.G + (c2.G-c1.G)*blend
            local rB = c1.B + (c2.B-c1.B)*blend
            r = r .. string.format('<font color="rgb(%d,%d,%d)">%s</font>', rR*255, rG*255, rB*255, ch)
            index = index + 1
        else
            r = r .. ch
        end
    end
    return r
end

local function SafeRequest(requestData)
    local success, result = pcall(function()
        if syn and syn.request then
            return syn.request(requestData)
        elseif request and type(request) == "function" then
            return request(requestData)
        elseif http and http.request then
            return http.request(requestData)
        else
            local body = HttpService:GetAsync(requestData.Url)
            return { Body = body, StatusCode = 200, Success = true }
        end
    end)
    if success then return result
    else return { Body = "{}", StatusCode = 0, Success = false } end
end

local function RetryRequest(requestData, retries)
    retries = retries or 2
    for i = 1, retries do
        local result = SafeRequest(requestData)
        if result and result.StatusCode == 200 then return result end
        task.wait(1)
    end
    return { Success = false }
end

local function findTycoon()
    for _, v in pairs(workspace:GetChildren()) do
        if v:IsA("Folder") and v.Name:match("Tycoon%d") then
            if v:FindFirstChild("Owner") and v.Owner.Value == LocalPlayer then
                return v
            end
        end
    end
end

local function _refreshTycoon()
    local t = findTycoon()
    if t and t ~= userTycoon then userTycoon = t end
    return userTycoon
end

task.spawn(function()
    while task.wait(3) do
        if not userTycoon or not userTycoon.Parent then _refreshTycoon() end
    end
end)

    if c:IsA("Folder") and c.Name:match("^Tycoon%d") then
        task.wait(0.5)
        _refreshTycoon()
    end
end)

local function withTycoon(fn)
    if not userTycoon or not userTycoon.Parent then
        if not _refreshTycoon() then return end
    end
    local ok, err = pcall(fn, userTycoon)
    if not ok then
        _refreshTycoon()
    end
    return ok, err
end

local _rateLimits = {}
local function safeInvoke(remote, cooldown, ...)
    if not remote or not remote.Parent then return false, "no remote" end
    local key = remote:GetDebugId(0)
    local now = tick()
    if _rateLimits[key] and now - _rateLimits[key] < (cooldown or 0.05) then
        return false, "rate"
    end
    _rateLimits[key] = now
    local args = { ... }
    local ok, res = pcall(function() return remote:InvokeServer(unpack(args)) end)
    return ok, res
end

local function _asyncFarm(startName, endName, flagFn, cooldown)
    task.spawn(function()
        while task.wait(1) do
                if flagFn() then
                local s = _remoteRequest(startName)
                local e = _remoteRequest(endName)
                if s and e then
                    task.spawn(function()
                        pcall(function() s:InvokeServer() end)
                    end)
                    task.wait(2)
                    pcall(function() e:InvokeServer() end)
                end
                task.wait(cooldown)
            end
        end
    end)
end

local function _stopFly()

    _stopFly()
    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    _flyBv = Instance.new("BodyVelocity")
    _flyBv.MaxForce = Vector3.new(math.huge, math.huge, math.huge)
    _flyBv.Velocity = Vector3.zero
    _flyBv.Parent = hrp
    _flyBg = Instance.new("BodyGyro")
    _flyBg.MaxTorque = Vector3.new(math.huge, math.huge, math.huge)
    _flyBg.P = 9000
    _flyBg.CFrame = hrp.CFrame
    _flyBg.Parent = hrp
    _flyConn = RunService.RenderStepped:Connect(function()
        if not FlyEnabled then _stopFly(); return end
        local cam = workspace.CurrentCamera
        local dir = Vector3.zero
        if UIS:IsKeyDown(Enum.KeyCode.W) then dir = dir + cam.CFrame.LookVector end
        if UIS:IsKeyDown(Enum.KeyCode.S) then dir = dir - cam.CFrame.LookVector end
        if UIS:IsKeyDown(Enum.KeyCode.A) then dir = dir - cam.CFrame.RightVector end
        if UIS:IsKeyDown(Enum.KeyCode.D) then dir = dir + cam.CFrame.RightVector end
        if UIS:IsKeyDown(Enum.KeyCode.Space) then dir = dir + Vector3.new(0,1,0) end
        if UIS:IsKeyDown(Enum.KeyCode.LeftControl) then dir = dir - Vector3.new(0,1,0) end
        if _flyBv and _flyBv.Parent then
            _flyBv.Velocity = dir.Magnitude > 0 and dir.Unit * FlySpeed or Vector3.zero
        end
        if _flyBg and _flyBg.Parent then _flyBg.CFrame = cam.CFrame end
    end)
end

local _LEVER_COLORS = {
    Red    = Color3.fromRGB(255, 60, 60),
    Purple = Color3.fromRGB(180, 60, 255),
    Green  = Color3.fromRGB(60, 255, 60),
    Blue   = Color3.fromRGB(60, 130, 255),
}

local _espSeeded = false

local function _matchLever(d)
    if not d.Name:lower():match("lever") then return end
    if not (d:IsA("Model") or d:IsA("BasePart")) then return end
    local col = Color3.fromRGB(255,255,255)
    for name, c in pairs(_LEVER_COLORS) do
        if d:GetFullName():find(name) then col = c; break end
    end
    return d.Name, col
end

local function _matchVine(d)
    if not d.Name:lower():match("vine") then return end
    if not (d:IsA("Model") or d:IsA("BasePart")) then return end
    return "Vine", Color3.fromRGB(100, 255, 100)
end

local function _matchCash(d)
    if not d:IsA("BasePart") then return end
    if not d.Name:lower():match("cash") then return end
    if not d:IsDescendantOf(workspace:FindFirstChild("CashDrops") or workspace) then return end
    return "$ Cash", Color3.fromRGB(255, 215, 0)
end

    if EspLevers then
        local lbl, col = _matchLever(d)
        if lbl then _addEsp(d, lbl, col); return end
    end
    if EspVines then
        local lbl, col = _matchVine(d)
        if lbl then _addEsp(d, lbl, col); return end
    end
    if EspCashDrops then
        local lbl, col = _matchCash(d)
        if lbl then _addEsp(d, lbl, col); return end
    end
end

    if _espSeeded then return end

        pcall(function() data.bb:Destroy() end)
        _esp[d] = nil
    end
end)

do
    local _buyCache = { list = {}, at = 0, tycoon = nil }
    local function _rebuildBuyCache(tycoon)
        local out = {}
        local purchases = tycoon:FindFirstChild("Purchases")
        if not purchases then _buyCache.list = out; _buyCache.at = tick(); _buyCache.tycoon = tycoon; return end
        for _, obj in ipairs(purchases:GetDescendants()) do
            if obj:IsA("Model") then
                local purchase = obj:FindFirstChild("Purchase")
                if purchase and purchase:IsA("RemoteFunction") then
                    local price = obj:GetAttribute("Price") or obj:GetAttribute("Cost") or 0
                    out[#out+1] = { model = obj, remote = purchase, price = tonumber(price) or 0 }
                end
            end
        end
        table.sort(out, function(a, b) return a.price < b.price end)
        _buyCache.list = out
        _buyCache.at = tick()
        _buyCache.tycoon = tycoon
    end
    local function buyAllAffordable()
        withTycoon(function(t)
            if _buyCache.tycoon ~= t or tick() - _buyCache.at > 2 then
                _rebuildBuyCache(t)
            end
            for _, entry in ipairs(_buyCache.list) do
                local m = entry.model
                if m.Parent and m:GetAttribute("Shown") == true and m:GetAttribute("Purchased") ~= true then
                    local ok = safeInvoke(entry.remote, 0.5)
                    if ok then stats.buys = stats.buys + 1 end
                end
            end
        end)
    end
    task.spawn(function()
        while true do
            task.wait(0.2)
            if AutoBuy then buyAllAffordable() end
        end
    end)
end

local upgradeRemotes = {}

local upgradeFails = {}

local function refreshUpgradeRemotes()
    upgradeRemotes = {}
    upgradeLevel = {}
    upgradeFails = {}
    if not userTycoon or not userTycoon.Parent then return end
    local purchases = userTycoon and userTycoon.Parent and userTycoon:FindFirstChild("Purchases")
    if not purchases then return end
    for _, obj in ipairs(purchases:GetDescendants()) do
        if obj:IsA("RemoteFunction") and obj.Name == "Upgrade" then
            upgradeRemotes[#upgradeRemotes + 1] = obj
        end
    end
    _lastScannedTycoon = userTycoon
end

task.spawn(function()
    while true do
        task.wait(0.25)
        if AutoUpgrade then
            if _lastScannedTycoon ~= userTycoon or tick() - lastUpgradeScan > 3 then
                refreshUpgradeRemotes()
                lastUpgradeScan = tick()
            end
            for _, remote in ipairs(upgradeRemotes) do
                if remote.Parent and (upgradeFails[remote] or 0) < 5 then
                    local lvl = (upgradeLevel[remote] or 0) + 1
                    local perTick = 0
                    while lvl <= 100 and perTick < 8 do
                        local ok, res = pcall(function() return remote:InvokeServer(lvl) end)
                        if (not ok) or res == false then
                            upgradeFails[remote] = (upgradeFails[remote] or 0) + 1
                            break
                        end
                        upgradeFails[remote] = 0
                        upgradeLevel[remote] = lvl
                        stats.upgrades = stats.upgrades + 1
                        lvl = lvl + 1
                        perTick = perTick + 1
                    end
                end
            end
        end
    end
end)

local NUM_SCALE = {
    thousand=1e3, million=1e6, billion=1e9, trillion=1e12, quadrillion=1e15,
    quintillion=1e18, sextillion=1e21, septillion=1e24, octillion=1e27,
    nonillion=1e30, decillion=1e33, undecillion=1e36, duodecillion=1e39,
    tredecillion=1e42, quattuordecillion=1e45, quindecillion=1e48,
    sexdecillion=1e51, septendecillion=1e54, octodecillion=1e57,
    novemdecillion=1e60, vigintillion=1e63,
    k=1e3, m=1e6, b=1e9, t=1e12, qd=1e15, qn=1e18, sx=1e21, sp=1e24,
}

do
    local BASE = {
        [0]="thousand",[1]="million",[2]="billion",[3]="trillion",[4]="quadrillion",
        [5]="quintillion",[6]="sextillion",[7]="septillion",[8]="octillion",[9]="nonillion",
        [10]="decillion",[11]="undecillion",[12]="duodecillion",[13]="tredecillion",
        [14]="quattuordecillion",[15]="quindecillion",[16]="sexdecillion",
        [17]="septendecillion",[18]="octodecillion",[19]="novemdecillion",
    }
    local ROOT = {
        [2]="vigintillion",[3]="trigintillion",[4]="quadragintillion",[5]="quinquagintillion",
        [6]="sexagintillion",[7]="septuagintillion",[8]="octogintillion",[9]="nonagintillion",
        [10]="centillion",
    }
    local PREFIX = {
        [0]="",[1]="un",[2]="duo",[3]="tres",[4]="quattuor",[5]="quin",
        [6]="sex",[7]="septen",[8]="octo",[9]="novem",
    }
    for n = 0, 100 do
        local name
        if n < 20 then
            name = BASE[n]
        else
            name = PREFIX[n % 10] .. ROOT[n // 10]
        end
        if name and name ~= "" then NUM_SCALE[name] = 10 ^ ((n + 1) * 3) end
    end
end

local function parseNumber(s)
    if not s then return nil end
    s = tostring(s):gsub(",", ""):lower()
    local num = s:match("[%d%.]+")
    local val = num and tonumber(num)
    if not val then return nil end
    local word = s:match("[%d%.%s]+([a-z]+)")
    if word and NUM_SCALE[word] then val = val * NUM_SCALE[word] end
    return val
end

local function getInvestorLogs()
    local bal = comp(CompClass.Balances)
    local reb = comp(CompClass.Rebirth)
    if not (bal and reb) then return nil end
    local okp, p = pcall(function() return reb:GetPotentialInvestors() end)
    local okc, c = pcall(function() return bal:GetInvestors() end)
    if okp and okc and type(p) == "number" and type(c) == "number" then
        return p, c
    end
    return nil
end

task.spawn(function()
    while true do
        task.wait(0.5)
        if AutoRebirth and not rebirthBusy then
            local remote = getRebirthRemote()
            local mult   = RebirthMultMode and RebirthInvestorMultiple or RebirthGainMultiple
            local worthIt = false
            local potLog, curLog = getInvestorLogs()
            if remote and potLog then
                worthIt = potLog >= math.log10(MinPotential)
                    and potLog >= curLog + math.log10(mult)
            elseif remote then
                local potential = getPotentialInvestors()
                local current   = getCurrentInvestors()
                worthIt = potential ~= nil
                    and potential >= MinPotential
                    and potential >= current * mult
            end
            if worthIt then
                rebirthBusy = true
                pcall(function()
                    local done   = false
                    local signal = getRebirthedSignal()
                    local conn
                    if signal and signal:IsA("RemoteEvent") then
                        conn = signal.OnClientEvent:Connect(function() done = true end)
                    end
                    remote:InvokeServer()
                    stats.rebirths = stats.rebirths + 1
                    local t = 0
                    while not done and t < RebirthTimeout do
                        task.wait(0.1)
                        t = t + 0.1
                    end
                    if conn then conn:Disconnect() end
                end)
                task.wait(RebirthCooldown)
                rebirthBusy = false
            end
        end
    end
end)

local function pullAllLevers()
    local char = LocalPlayer.Character
    local hrp  = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return 0 end
    local map   = workspace:FindFirstChild("Map")
    local sewer = map and map:FindFirstChild("Sewer")
    local root  = sewer or workspace
    local pulled = 0
    for _, o in ipairs(root:GetDescendants()) do
        if o:IsA("BasePart") and (o.Name == "Lever" or string.find(string.lower(o.Name), "lever", 1, true)) then
            pcall(function()
                firetouchinterest(hrp, o, 0)
                firetouchinterest(hrp, o, 1)
            end)
            pulled = pulled + 1
        end
    end
    if sewer then
        for _, o in ipairs(sewer:GetDescendants()) do
            if o:IsA("BasePart") and (o.Name == "VineKey" or o.Name == "UFOKey") then
                pcall(function()
                    firetouchinterest(hrp, o, 0)
                    firetouchinterest(hrp, o, 1)
                end)
            end
        end
    end
    return pulled
end

local function doSewerRun()
    local char = LocalPlayer.Character
    local hrp  = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return false, "no character" end
    local map   = workspace:FindFirstChild("Map")
    local sewer = map and map:FindFirstChild("Sewer")
    if not sewer then return false, "sewer not loaded" end
    for _, o in ipairs(sewer:GetDescendants()) do
        if o:IsA("BasePart") and string.find(string.lower(o.Name), "lever", 1, true) then
            touchPart(hrp, o)
        end
    end
    for _, folderName in ipairs({ "CashVine", "SewerAlien" }) do
        local folder = sewer:FindFirstChild(folderName)
        if folder then
            for _, o in ipairs(folder:GetDescendants()) do
                if o:IsA("BasePart") and (o.Name == "VineKey" or o.Name == "UFOKey") then
                    touchPart(hrp, o)
                end
            end
        end
    end
    task.wait(0.3)
    local cashVine = sewer:FindFirstChild("CashVine")
    if cashVine then
        local vineDoor = cashVine:FindFirstChild("VineDoor")
        if vineDoor then
            for _, o in ipairs(vineDoor:GetDescendants()) do
                if o:IsA("BasePart") then touchPart(hrp, o) end
            end
        end
    end
    task.wait(0.3)
    if cashVine then
        local vineModel = cashVine:FindFirstChild("CashVine")
        if vineModel then
            local pivot = vineModel:GetPivot()
            pcall(function() hrp.CFrame = pivot + Vector3.new(0, 3, 0) end)
            task.wait(0.2)
            for _, o in ipairs(vineModel:GetDescendants()) do
                if o:IsA("BasePart") then touchPart(hrp, o) end
            end
        end
    end
    return true
end

local FRUIT_SETTLE = 0.1
local function collectFruit(tree)
    local detectors, positions = {}, {}
    for _, obj in ipairs(tree:GetDescendants()) do
        if obj:IsA("BasePart") and obj.Name == "Fruit" then
            obj.CanCollide = false
            local clickPart = obj:FindFirstChild("ClickFruitPart") or obj:FindFirstChild("ClickPart")
            if clickPart then
                local cd = clickPart:FindFirstChildOfClass("ClickDetector")
                if cd then
                    detectors[#detectors+1] = cd
                    positions[#positions+1] = clickPart.Position
                end
            end
        end
    end
    local n = #detectors
    if n == 0 then return end

    noCollisionTree(tree)

    local char = LocalPlayer.Character
    local hrp = char and char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    for i = 1, n do
        if not AutoFruit then return end
        hrp.CFrame = CFrame.new(positions[i] + Vector3.new(0, 3, 0))
        task.wait(FRUIT_SETTLE)
        pcall(fireclickdetector, detectors[i])
        stats.fruit = stats.fruit + 1
    end
end

do
    local _touchCache = {}
    local _touchSeeded = {}
    local function _seedTouchCache(cat, pred)
        if _touchSeeded[cat] then return end
        _touchSeeded[cat] = true
        _touchCache[cat] = {}
        for _, o in ipairs(workspace:GetDescendants()) do
            if pred(o) then _touchCache[cat][o] = true end
        end
    end
    local _touchPredicates = {
        cosmicFarmer = function(o) return o:IsA("BasePart") and (o.Name:match("Cosmic") or o.Name:match("Farmer")) end,
        hiddenBadge  = function(o) return o:IsA("BasePart") and (o.Name:match("Badge") or o.Name:match("Secret") or o.Name:match("Hidden")) end,
        anyBadge     = function(o) return o:IsA("BasePart") and o.Name:match("Badge") end,
    }
    workspace.DescendantAdded:Connect(function(d)
        for cat, pred in pairs(_touchPredicates) do
            if _touchSeeded[cat] and pred(d) then _touchCache[cat][d] = true end
        end
    end)
    workspace.DescendantRemoving:Connect(function(d)
        for _, tbl in pairs(_touchCache) do if tbl[d] then tbl[d] = nil end end
    end)
    local function _touchAll(cat)
        _seedTouchCache(cat, _touchPredicates[cat])
        local char = LocalPlayer.Character
        local hrp  = char and char:FindFirstChild("HumanoidRootPart")
        if not hrp then return end
        for o in pairs(_touchCache[cat]) do
            if o.Parent then
                pcall(function()
                    firetouchinterest(hrp, o, 0)
                    firetouchinterest(hrp, o, 1)
                end)
            end
        end
    end
    task.spawn(function() while true do task.wait(3); if AutoCosmicFarmer then _touchAll("cosmicFarmer") end end end)
    task.spawn(function() while true do task.wait(5); if AutoHiddenBadges then _touchAll("hiddenBadge") end end end)
    task.spawn(function() while true do task.wait(5); if AutoCollectBadges then _touchAll("anyBadge") end end end)
end
